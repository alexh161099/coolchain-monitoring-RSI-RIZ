12def check_stimmigkeit(rows):
13 """
14 rows: Liste von Tupeln (station_id, direction, dt)
15 Prüft pro Station die korrekte Reihenfolge: immer 'in' gefolgt von 'out'.
16 'in' am Ende (Transport noch nicht abgeschlossen) ist erlaubt.
17 Rückgabe: (ok: bool, meldung: str)
18 """
19 per_station = {}
20 for station_id, direction, dt in rows:
21 per_station.setdefault(station_id, []).append((direction, dt))
22
23 for station_id, events in per_station.items():
24 last = None
25 for direction, _ in events:
26 if direction not in ("in", "out"):
27 return (False, f"Unbekannte Richtung '{direction}' bei Station {station_id}.")
28 if last is None and direction != "in":
29 return (False, f"Erster Eintrag bei Station {station_id} ist nicht 'in'.")
30 if last == direction:
31 return (False, f"Doppelte Richtung '{direction}' nacheinander bei Station {station_id}.")
32 last = direction
33 # Hinweis: endet eine Station mit 'in', ist das zulässig (Transport evtl. noch offen).
34
35 return (True, "Stimmigkeit ok.")

39 """
40 Prüft, dass zwischen einem 'out' und dem darauffolgenden 'in'
41 höchstens 10 Minuten liegen.
42 Rückgabe: (ok: bool, meldung: str)
43 """
44 prev_out = None
45 for _station_id, direction, dt in rows:
46 if direction == "out":
47 prev_out = dt
48 elif direction == "in" and prev_out is not None:
49 delta = dt - prev_out
50 if delta > timedelta(minutes=10):
51 mins = math.ceil(delta.total_seconds() / 60.0) # aufrunden für klare Anzeige
52 return (False, f"Übergabe > 10 min: {mins} min.")
53 prev_out = None
54 return (True, "Übergaben ≤ 10 min.")
